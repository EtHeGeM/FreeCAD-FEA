\textbf{Original question visual:}\par

\textbf{Step-by-step solution:}\par
\textbf{Given:} $m=40\,\mathrm{kg}$, $k=800\,\mathrm{N/m}$, $F=100\cos(2t)\,\mathrm{N}$.
\begin{enumerate}
\item FBD: spring force $kx$ acts opposite displacement; harmonic force acts downward; use $x$ from static equilibrium.
\item Equation: $mx''+kx=F_0\cos\omega t$.
\item $\omega_n=\sqrt{k/m}=\sqrt{800/40}=4.472\,\mathrm{rad/s}$.
\item Steady-state amplitude, for $\omega\neq\omega_n$:
\[X={F_0\over |k-m\omega^2|}={100\over |800-40(2)^2|}=0.15625\,\mathrm{m}\]
\item Velocity: $v=-\omega X\sin\omega t$.
\[\boxed{v_{max}=\omega X=0.3125\,\mathrm{m/s}}\]
\end{enumerate}
